def self.create(content_hash, size)
  content = Content.new(content_hash: content_hash, size: size, crypt_key: WodaCrypt.new.random_key.to_hex)
  i = 0
  while size > 0 do
    content.x_parts << XPart.new(part_number: i)
    size -= PART_SIZE
    i += 1
  end
  content.save
  content
end
